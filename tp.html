import sympy as sp
import matplotlib.pyplot as plt

# Définir les variables symboliques
x, y = sp.symbols('x y')

# Définir les trois points non alignés
p1 = (2, 3)
p2 = (5, 7)
p3 = (-1, 4)

# Créer les équations pour trouver le centre du cercle
eq1 = sp.Eq((x - p1[0])**2 + (y - p1[1])**2, (x - p2[0])**2 + (y - p2[1])**2)
eq2 = sp.Eq((x - p1[0])**2 + (y - p1[1])**2, (x - p3[0])**2 + (y - p3[1])**2)

# Résoudre les équations pour trouver le centre du cercle
center = sp.solve((eq1, eq2), (x, y))

# Calculer le rayon du cercle
radius = sp.sqrt((p1[0] - center[x])**2 + (p1[1] - center[y])**2)

# Afficher les résultats
print("Centre du cercle:", center)
print("Rayon du cercle:", radius)

# Plot the circle and points
circle = plt.Circle((center[x], center[y]), radius, color='r', fill=False)
fig, ax = plt.subplots()
ax.add_artist(circle)
plt.plot(*zip(p1, p2, p3), marker='o', linestyle='None', color='b')

# Ajouter les labels
plt.xlabel('X')
plt.ylabel('Y')
plt.title('Cercle passant par trois points non alignés')

plt.axis('equal')
plt.show()
